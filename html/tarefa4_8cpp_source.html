24bool hit_triangle(const vec3& v0, const vec3& v1, const vec3& v2, const Ray& r) {
25 vec3 e1 = v1 - v0;
26 vec3 e2 = v2 - v0;
27 vec3 h = cross(r.direction(), e2);
28 double a = dot(e1, h);
29
30 if (a > -1e-6 && a < 1e-6)
31 return false;
32
33 double f = 1.0 / a;
34 vec3 s = r.origin() - v0;
35 double u = f * dot(s, h);
36
37 if (u < 0.0 || u > 1.0)
38 return false;
39
40 vec3 q = cross(s, e1);
41 double v = f * dot(r.direction(), q);
42
43 if (v < 0.0 || u + v > 1.0)
44 return false;
45
46 double t = f * dot(e2, q);
47
48 return t > 1e-6; // Considerar apenas interse칞칫es frente do raio
49}

59bool hit_sphere(const vec3& center, double radius, const Ray& r) {
60 vec3 oc = r.origin() - center;
61 auto a = dot(r.direction(), r.direction());
62 auto b = 2.0 * dot(oc, r.direction());
63 auto c = dot(oc, oc) - radius * radius;
64 auto discriminant = b * b - 4 * a * c;
65 return (discriminant >= 0);
66}
